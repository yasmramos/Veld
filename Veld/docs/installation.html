üì¶ Maven Installation

Basic Maven Setup

Add Veld dependencies to your pom.xml:

<dependencies>
    <!-- Veld Core Runtime -->
    <dependency>
        <groupId>io.github.yasmramos.veld</groupId>
        <artifactId>veld-runtime</artifactId>
        <version>1.0.2</version>
    </dependency>
    
    <!-- Veld Annotations -->
    <dependency>
        <groupId>io.github.yasmramos.veld</groupId>
        <artifactId>veld-annotations</artifactId>
        <version>1.0.2</version>
    </dependency>
</dependencies>

Annotation Processor Configuration

<build>
    <plugins>
        <plugin>
            <groupId>org.apache.maven.plugins</groupId>
            <artifactId>maven-compiler-plugin</artifactId>
            <version>3.11.0</version>
            <configuration>
                <source>11</source>
                <target>11</target>
                <annotationProcessorPaths>
                    <path>
                        <groupId>io.github.yasmramos.veld</groupId>
                        <artifactId>veld-processor</artifactId>
                        <version>1.0.2</version>
                    </path>
                </annotationProcessorPaths>
            </configuration>
        </plugin>
    </plugins>
</build>

Complete Maven Configuration

Full Maven setup with all Veld modules:

<project>
    <modelVersion>4.0.0</modelVersion>
    <groupId>com.example</groupId>
    <artifactId>my-veld-app</artifactId>
    <version>1.0.0</version>
    
    <properties>
        <maven.compiler.source>11</maven.compiler.source>
        <maven.compiler.target>11</maven.compiler.target>
        <project.build.sourceEncoding>UTF-8</project.build.sourceEncoding>
        <veld.version>1.0.2</veld.version>
    </properties>
    
    <dependencies>
        <!-- Veld Core -->
        <dependency>
            <groupId>io.github.yasmramos.veld</groupId>
            <artifactId>veld-runtime</artifactId>
            <version>${veld.version}</version>
        </dependency>
        
        <dependency>
            <groupId>io.github.yasmramos.veld</groupId>
            <artifactId>veld-annotations</artifactId>
            <version>${veld.version}</version>
        </dependency>
        
        <!-- Optional Veld Modules -->
        <dependency>
            <groupId>io.github.yasmramos.veld</groupId>
            <artifactId>veld-aop</artifactId>
            <version>${veld.version}</version>
        </dependency>
        
        <dependency>
            <groupId>io.github.yasmramos.veld</groupId>
            <artifactId>veld-weaver</artifactId>
            <version>${veld.version}</version>
        </dependency>
    </dependencies>
    
    <build>
        <plugins>
            <plugin>
                <groupId>org.apache.maven.plugins</groupId>
                <artifactId>maven-compiler-plugin</artifactId>
                <version>3.11.0</version>
                <configuration>
                    <source>11</source>
                    <target>11</target>
                    <annotationProcessorPaths>
                        <path>
                            <groupId>io.github.yasmramos.veld</groupId>
                            <artifactId>veld-processor</artifactId>
                            <version>${veld.version}</version>
                        </path>
                    </annotationProcessorPaths>
                    <compilerArgs>
                        <arg>-Amapstruct.defaultComponentModel=veld</arg>
                    </compilerArgs>
                </configuration>
            </plugin>
            
            <plugin>
                <groupId>org.apache.maven.plugins</groupId>
                <artifactId>maven-surefire-plugin</artifactId>
                <version>3.0.0</version>
                <configuration>
                    <argLine>
                        --add-opens java.base/java.lang=ALL-UNNAMED
                    </argLine>
                </configuration>
            </plugin>
        </plugins>
    </build>
</project>

‚öôÔ∏è Gradle Installation

Basic Gradle Setup

Add Veld to your build.gradle:

plugins {
    id 'java'
    id 'application'
}

group = 'com.example'
version = '1.0.0'

java {
    sourceCompatibility = JavaVersion.VERSION_11
    targetCompatibility = JavaVersion.VERSION_11
}

repositories {
    mavenCentral()
}

dependencies {
    implementation 'io.github.yasmramos.veld:veld-runtime:1.0.2'
    implementation 'io.github.yasmramos.veld:veld-annotations:1.0.2'
    
    // Annotation processor
    annotationProcessor 'io.github.yasmramos.veld:veld-processor:1.0.2'
    
    // Test dependencies
    testImplementation 'org.junit.jupiter:junit-jupiter:5.9.0'
    testImplementation 'org.mockito:mockito-core:5.1.0'
}

Annotation Processor Configuration

compileJava {
    options.annotationProcessorPath = configurations.annotationProcessor
    options.compilerArgs += [
        '-Amapstruct.defaultComponentModel=veld',
        '-Aveld.optimization.enabled=true'
    ]
}

Gradle with Kotlin DSL

For build.gradle.kts:

plugins {
    java
    application
}

group = "com.example"
version = "1.0.0"

java {
    sourceCompatibility = JavaVersion.VERSION_11
    targetCompatibility = JavaVersion.VERSION_11
}

repositories {
    mavenCentral()
}

dependencies {
    implementation("io.github.yasmramos.veld:veld-runtime:1.0.2")
    implementation("io.github.yasmramos.veld:veld-annotations:1.0.2")
    
    // Annotation processor
    annotationProcessor("io.github.yasmramos.veld:veld-processor:1.0.2")
}

tasks.withType<JavaCompile> {
    options.annotationProcessorPath = configurations["annotationProcessor"]
    options.compilerArgs.addAll(listOf(
        "-Amapstruct.defaultComponentModel=veld",
        "-Aveld.optimization.enabled=true"
    ))
}

üî® Build Tools Integration

Ant with Ivy

<!-- build.xml -->
<project>
    <property name="lib.dir" value="lib"/>
    
    <path id="veld.classpath">
        <fileset dir="${lib.dir}">
            <include name="veld-runtime*.jar"/>
            <include name="veld-annotations*.jar"/>
        </fileset>
    </path>
    
    <target name="compile">
        <mkdir dir="build/classes"/>
        <javac srcdir="src/main/java"
              destdir="build/classes"
              classpathref="veld.classpath"
              compiler="modern">
            <compilerclasspath>
                <fileset dir="${lib.dir}">
                    <include name="velocity*.jar"/>
                </fileset>
            </compilerclasspath>
        </javac>
    </target>
</project>

üñ•Ô∏è IDE Configuration

IntelliJ IDEA

IntelliJ IDEA provides excellent support for Veld out of the box:

Project Structure

  • Enable annotation processing: Settings ‚Üí Build ‚Üí Compiler ‚Üí Annotation Processors
  • Enable incremental compilation for better performance
  • Configure annotation processor path to include Veld processor

Annotation Processor Settings

  • Obtain processors from project classpath: Enabled
  • Processor path: Include veld-processor JAR
  • Supported options: All Veld processor options

Configuration File

# .idea/compiler.xml
<component name="CompilerConfiguration">
    <annotationProcessing>
        <profile default="true" name="Default" enabled="true">
            <processorPath useClasspath="false">
                <entry name="io.github.yasmramos.veld:veld-processor:1.0.2"/>
            </processorPath>
        </profile>
    </annotationProcessing>
</component>

Eclipse

Configure Eclipse for annotation processing:

Project Properties

  • Right-click project ‚Üí Properties
  • Navigate to Java Compiler ‚Üí Annotation Processing
  • Enable annotation processing
  • Add Veld processor to factory path

Factory Path Configuration

<factorypath>
    <factorypathentry kind="VARJAR" 
                      id="io.github.yasmramos.veld:veld-processor:1.0.2" 
                      enabled="true" 
                      runInBatchMode="false"/>
</factorypath>

Test Installation

Create a simple test to verify Veld is working:

// src/main/java/com/example/TestComponent.java
package com.example;

import io.github.yasmramos.veld.annotation.Component;
import io.github.yasmramos.veld.annotation.Inject;

@Component
public class TestComponent {
    
    @Inject
    private TestService testService;
    
    public void test() {
        System.out.println("Veld is working! " + testService.getMessage());
    }
}

// src/main/java/com/example/TestService.java
package com.example;

import io.github.yasmramos.veld.annotation.Component;

@Component
public class TestService {
    public String getMessage() {
        return "Hello from Veld!";
    }
}

// src/main/java/com/example/Application.java
package com.example;

import io.github.yasmramos.veld.runtime.Veld;

public class Application {
    public static void main(String[] args) {
        // Use the generated Veld class;
        container.start();
        
        TestComponent component = container.getBean(TestComponent.class);
        component.test();
        
        container.shutdown();
    }
}

Troubleshooting

Common Issues

  • Annotation processor not found: Check processor path configuration
  • Generated files missing: Verify annotation processing is enabled
  • Compilation errors: Ensure Java version compatibility
  • IDE not recognizing Veld annotations: Check IDE annotation processing settings

Debug Mode

# Enable debug logging
# Maven
mvn clean compile -X

# Gradle
./gradlew clean compileJava --info
