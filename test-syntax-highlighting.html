Java Code Example

package com.example.service;

import com.veld.annotation.Component;
import com.veld.annotation.Inject;
import com.veld.annotation.Value;

@Component
public class UserService {
    
    @Inject
    private DatabaseService database;
    
    @Value("${app.name:VeldApp}")
    private String appName;
    
    public User getUser(String id) {
        // Query the database
        return database.findById(id);
    }
    
    public void createUser(User user) {
        database.save(user);
        System.out.println("User created: " + user.getName());
    }
}

import com.veld.runtime.VeldContainer;

public class Main {
    public static void main(String[] args) {
        VeldContainer container = new VeldContainer();
        container.scan("com.example");
        container.start();
        
        UserService userService = container.getBean(UserService.class);
        userService.createUser(new User("John", "john@example.com"));
    }
}
